172 void RobotinoSafety::inE2(geometry_msgs::Point32 point)
173 {
174  double check = pow( (point.x / e2_major_radius_), 2 ) + pow( (point.y / e2_minor_radius_), 2 ) - 1;
175 
176  if( check <= 0.0 ) // Check if the point is in Ellipse 2
177  {
178  slow_laser_ = true;
179 
180  dist_ = solveE1(point);
181  if(dist_ <= 0.0 )
182  {
183  dist_ = 0.0;

185  }
