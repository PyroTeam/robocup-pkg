49 void KinectROS::init()
50 {
51  for (size_t i=0; i < 2048; i++)
52  {
53  const float k1 = 1.1863;
54  const float k2 = 2842.5;
55  const float k3 = 0.1236;
56  gamma_[i] = k3 * tan(i/k2 + k1);
57  }
58 
59  Eigen::Matrix4f transformationX; // Rotate 90 Degrees along X
60  transformationX <<
61  1, 0, 0, 0,
62  0, 0, 1, 0,
63  0, -1, 0, 0,
64  0, 0, 0, 1;
65  Eigen::Matrix4f transformationY; // Rotate 90 Degrees along Z
66  transformationY <<
67  0, 0, 1, 0,
68  0, 1, 0, 0,
69  -1, 0, 0, 0,
70  0, 0, 0, 1;
71 
72  transformation_ = transformationX * transformationY;
73 }

88  // Build the cloud message
89  msg->header.stamp = stamp_;
90  msg->header.frame_id = "kinect_link";
91  msg->height = height;
92  msg->width = width;
93  msg->is_dense = false;
94 
95  msg->points.resize( msg->height * msg->width );
96 
97  // camera intrinsic parameters, representative values, see http://nicolas.burrus.name/index.php/Research/KinectCalibration for more info
98  float cx = 320.0; // center of projection
99  float cy = 240.0; // center of projection
100  float fx = 600.0; // focal length in pixels
101  float fy = 600.0; // focal length in pixels
102  for (size_t v=0, n=0 ; v<480 ; v++)
103  {
104  for (size_t u=0 ; u<640 ; u++, n++)
105  {
106  // note that values will be in meters, with the camera at the origin, and the Kinect has a minimum range of ~0.5 meters
107  msg->points[n].x = (u - cx) * gamma_[data[n]] / fx;
108  msg->points[n].y = (v - cy) * gamma_[data[n]] / fy;
109  msg->points[n].z = gamma_[data[n]];
110  if( msg->points[n].z < 0.2 ) //to close
111  {
112  msg->points[n].x = msg->points[n].y = msg->points[n].z =
113  std::numeric_limits<float>::quiet_NaN();
114  }
115  }
116  }
117 
118  pcl::transformPointCloud (*msg, msg_transformed, transformation_ );
119 
120  if( downsample_ )
121  {
122  sensor_msgs::PointCloud2 cloud;
123  sensor_msgs::PointCloud2::Ptr cloud_downsampled (new sensor_msgs::PointCloud2());
124  pcl::VoxelGrid<sensor_msgs::PointCloud2> vg;
125  pcl::toROSMsg( msg_transformed, cloud );
126  vg.setInputCloud( boost::make_shared<sensor_msgs::PointCloud2> (cloud) );
127  vg.setLeafSize (leaf_size_, leaf_size_, leaf_size_);
128  vg.filter (*cloud_downsampled);
129 
130  cloud_pub_.publish( cloud_downsampled );
131  }
132  else
133  {
134  cloud_pub_.publish( msg );
135  }
136 }
